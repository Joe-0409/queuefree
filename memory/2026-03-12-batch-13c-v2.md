```ts
import { applyDecorators } from '@nestjs/common';
import {
 ApiBadRequestResponse,
 ApiConflictResponse,
 ApiInternalServerErrorResponse,
 ApiNotFoundResponse
} from '@nestjs/swagger';
import { ApiErrorResponseDto } from '../dto/api-error-response.dto';

interface ApiCommonErrorResponsesOptions {
 badRequestDescription?: string;
 notFoundDescription?: string;
 conflictDescription?: string;
}

export function ApiCommonErrorResponses(
 options?: ApiCommonErrorResponsesOptions
): MethodDecorator & ClassDecorator {
 const decorators: Array<MethodDecorator | ClassDecorator> = [
 ApiInternalServerErrorResponse({
 description: 'Internal server error',
 type: ApiErrorResponseDto
 })
 ];

 if (options?.badRequestDescription) {
 decorators.push(
 ApiBadRequestResponse({
 description: options.badRequestDescription,
 type: ApiErrorResponseDto
 })
 );
 }

 if (options?.notFoundDescription) {
 decorators.push(
 ApiNotFoundResponse({
 description: options.notFoundDescription,
 type: ApiErrorResponseDto
 })
 );
 }

 if (options?.conflictDescription) {
 decorators.push(
 ApiConflictResponse({
 description: options.conflictDescription,
 type: ApiErrorResponseDto
 })
 );
 }

 return applyDecorators(...decorators);
}
```

```ts
import { OrderStatus } from '@queuefree/shared';
import { BadRequestException, Injectable, NotFoundException } from '@nestjs/common';
import { CreateOrderRequestDto } from './dto/create-order-request.dto';
import { CreateOrderResponseDto } from './dto/create-order-response.dto';

const DEMO_PRODUCT_ID = 'prod_demo_01';
const DEMO_SKU_ID = 'sku_demo_01';
const DEMO_ADDRESS_ID = 'addr_demo_01';

@Injectable()
export class OrdersService {
 createOrder(
 body: CreateOrderRequestDto,
 idempotencyKey?: string
 ): CreateOrderResponseDto {
 if (!this.isNonEmptyString(idempotencyKey)) {
 throw new BadRequestException({
 message: 'Idempotency-Key header is required.'
 });
 }

 if (
 !this.isNonEmptyString(body.productId) ||
 !this.isNonEmptyString(body.skuId) ||
 !this.isNonEmptyString(body.addressId) ||
 typeof body.quantity !== 'number' ||
 !Number.isFinite(body.quantity) ||
 body.quantity < 1
 ) {
 throw new BadRequestException({
 message: 'Invalid order creation payload.'
 });
 }

 if (
 body.productId !== DEMO_PRODUCT_ID ||
 body.skuId !== DEMO_SKU_ID ||
 body.addressId !== DEMO_ADDRESS_ID
 ) {
 throw new NotFoundException({
 message: 'Product, SKU, or address not found.'
 });
 }

 return {
 orderId: 'order_demo_01',
 status: OrderStatus.WAIT_PAY,
 productId: body.productId,
 skuId: body.skuId,
 quantity: body.quantity
 };
 }

 private isNonEmptyString(value: unknown): value is string {
 return typeof value === 'string' && value.trim().length > 0;
 }
}
```

```ts
import { OrderStatus } from '@queuefree/shared';
import {
 ConflictException,
 Injectable,
 NotFoundException
} from '@nestjs/common';
import { CreatePaymentIntentResponseDto } from './dto/create-payment-intent-response.dto';

@Injectable()
export class PaymentsService {
 createPaymentIntent(
 orderId: string,
 _idempotencyKey?: string
 ): CreatePaymentIntentResponseDto {
 const orderStatus = this.getDemoOrderStatus(orderId);

 if (!orderStatus) {
 throw new NotFoundException({
 message: 'Order not found.'
 });
 }

 if (orderStatus !== OrderStatus.WAIT_PAY) {
 throw new ConflictException({
 message: 'Order cannot create a payment intent in its current status.',
 details: {
 currentStatus: orderStatus
 }
 });
 }

 return {
 paymentIntentId: 'pi_demo_01',
 orderId,
 provider: 'mockpay',
 amountMinor: 129900,
 currencyCode: 'PHP',
 checkoutUrl: 'https://checkout.queuefree.com/payment-intents/pi_demo_01'
 };
 }

 private getDemoOrderStatus(orderId: string): OrderStatus | null {
 switch (orderId) {
 case 'order_demo_01':
 return OrderStatus.WAIT_PAY;
 case 'order_paid_01':
 return OrderStatus.PAID;
 default:
 return null;
 }
 }
}
```
